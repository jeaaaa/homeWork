  getRepository(entity) {
    let repository = this.connection.getRepository(entity);
    return {
      async add(data) {
        await repository.save(data);
        console.log(`${data} has been saved`);
      },

      async update(query, data) {
        let obj = await repository.findOne(query);
        let newObj = Object.assign(obj, data);
        await repository.save(newObj);
        console.log(`${data} has been saved`);
      },

      async remove(query) {
        let obj = await repository.findOne(query);
        await repository.remove(obj);
        console.log(`${query} has been removed`);
      },

      async find(query) {
        let list = await repository.find(query);
        return list;
      },

      async findOne(query) {
        let obj = await repository.findOne(query);
        return obj;
      },

      async count(query) {
        let count = await repository.count(query);
        return count;
      },

      async findAndcCount(query) {
        let [list, count] = await repository.findAndCount(query);
        return [list, count];
      },

      async execute(sql) {
        const rawData = await repository.query(sql);
        return rawData;
      },
    };
  }
}
